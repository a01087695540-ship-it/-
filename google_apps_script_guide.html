1. Google Apps Script 복사용 코드

구글 스프레드시트 스크립트 에디터에 그대로 붙여넣는 프로덕션 코드입니다.

Code.gs (Google Apps Script) javascript 
/**
 * AX 부스터 과정 사전 설문조사 - Google Spreadsheet 실시간 연동 스크립트
 * 
 * 기능: 설문조사 웹앱으로부터 데이터를 실시간 POST 수신하여
 * 지정된 스프레드시트의 첫 번째 시트에 자동 기입합니다.
 */

function doPost(e) {
  try {
    // 1. 전송 수신 데이터 검증 및 JSON 파싱
    if (!e || !e.postData || !e.postData.contents) {
      throw new Error("전송된 데이터가 비어 있습니다. (No payload found)");
    }
    
    var jsonString = e.postData.contents;
    var data = JSON.parse(jsonString);
    
    // 2. 현재 스크립트가 포함된 스프레드시트의 첫 번째 시트 자동 선택
    // (이렇게 설정하면 별도의 ID 입력 없이 가장 안전하게 탐색 가능합니다)
    var sheet;
    try {
      sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0];
    } catch (activeErr) {
      // 혹시라도 구글 드라이브 독립 실행형 스크립트로 만든 특수한 상황일 경우
    }
    
    if (!sheet) {
      // [독립형 스크립트 대안] getActiveSpreadsheet를 로드하지 못한 경우 아래에 실제 본인 스프레드시트의 ID를 입력해야 합니다.
      // 예시: url이 https://docs.google.com/spreadsheets/d/abc12345/edit 이면 ID는 abc12345 입니다.
      var spreadsheetId = "17B9qQHaAmTSekuT0Xibcu92JeErr7dDUj3qMlZ7R2DA";
      sheet = SpreadsheetApp.openById(spreadsheetId).getSheets()[0];
    }
    
    // 3. 스프레드시트가 완전히 비어있는 신규 문서라면 헤더 테이블 행 자동 생성
    if (sheet.getLastRow() === 0) {
      var headers = [
        "제출 시각", 
        "이름", 
        "AI 툴 (Q2)", 
        "AI 사용 빈도 (Q3)", 
        "AI 활용 분야 (Q4)", 
        "AI 활용 수준 (Q5)", 
        "담당 업무 (Q6)", 
        "자동화 희망 작업 (Q7)", 
        "소요 시간 (Q8)", 
        "자동화 상세 내용 (Q9)", 
        "교육 기대사항 (Q10)", 
        "자유 의견 (Q11)"
      ];
      sheet.appendRow(headers);
      
      // 가독성을 위한 헤더 행 고정 및 폰트 볼드 처리 지정
      var headerRange = sheet.getRange(1, 1, 1, headers.length);
      headerRange.setFontWeight("bold");
      headerRange.setBackground("#F3F4F6"); // 깔끔한 회색 음영
      headerRange.setHorizontalAlignment("center");
      sheet.setFrozenRows(1);
    }
    
    // 4. 설문 데이터 안전 추출 및 배열화
    var rowData = [
      data["제출시각"] || new Date().toLocaleString("ko-KR"),
      data["이름"] || "",
      data["AI툴"] || "",
      data["AI사용빈도"] || "",
      data["AI활용분야"] || "",
      data["AI활용수준"] || "",
      data["담당업무"] || "",
      data["자동화희망작업"] || "",
      data["소요시간"] || "",
      data["자동화상세"] || "",
      data["교육기대사항"] || "",
      data["자유의견"] || ""
    ];
    
    // 5. 시트 최하단에 행 추가 기입
    sheet.appendRow(rowData);
    
    // 정상 성공 응답 반환
    return ContentService.createTextOutput(JSON.stringify({ "status": "success" }))
                         .setMimeType(ContentService.MimeType.JSON);
                         
  } catch (error) {
    // 에러 발생 로그 작성 및 오류 응답 반환
    Logger.log("Error during submission: " + error.toString());
    return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": error.toString() }))
                         .setMimeType(ContentService.MimeType.JSON);
  }
}

2. 구글 스프레드시트 연동 방법 (초간단 5단계)

1

스프레드시트 생성

Google 스프레드시트 에 접속하여 새 스프레드시트를 빈 문서로 생성합니다.
스프레드시트의 파일 제목은 원하시는 이름(예: Lotte GRS AX Booster 사전설문 결과)으로 지정합니다.

2

앱스 스크립트(Apps Script) 편집기 열기

1. 상단 메뉴바에서 [확장 프로그램] ➔ [Apps Script]를 클릭하면 코딩용 메모장 웹 창이 열립니다.
2. 기본적으로 에디터에 적혀 있는 function myFunction() { ... } 코드를 드래그하여 전부 삭제해 빈 화면으로 만듭니다.
3. 본 가이드 맨 위의 [복사용 코드] 상자에서 코드를 전부 복사해 에디터에 붙여넣고 상단 디스크 디스크 모양의 저장 단추(컨트롤+S)를 클릭합니다.

3

웹앱으로 배포하기 매우 중요 ⭐

설문지 웹사이트가 사용자의 구글 시트에 실시간 쓰기 명령을 수행할 수 있게 웹 API를 배포해야 합니다.
1. 화면 오른쪽 상단의 파란색 [배포] ➔ [새 배포] 버튼 단추를 누릅니다.
2. 배포 설정창 좌측 상단의 톱니바퀴를 눌러 유형 값을 [웹앱 (Web App)]으로 골라 줍니다.
3. 다음과 같이 정확하게 셋팅하십시오:

설명 (Description): AX 설문 조사 실시간 연동
실행 대상 (Execute as): 나 (사용자 본인의 구글 이메일)
액세스 권한 (Who has access): 모든 사용자 (Anyone)

⚠️ [액세스 권한]을 '모든 사용자(Anyone)'로 열어주어야 실시간 API 연동 명령이 정상 구동됩니다. 사용자의 실제 구글 드라이브나 다른 시트에는 접근하지 않고 오직 해당 시트의 doPost 파일만 열게 되므로 안전합니다.

4. [배포] 버튼을 누르시면 중간에 구글 측 확인 창과 더불어 "액세스 승인(Authorize Access)" 팝업이 발생합니다.
5. 구글 계정을 선택한 후, 경고창 하단의 작은 글씨인 [고급] ➔ [Go to 제목없는 프로젝트 (안전하지 않음)] ➔ [허용] 순서로 허가를 넘어가 주시면 됩니다.

4

생성된 웹앱 URL 복사

배포 처리가 최종 마무리되면 눈앞에 웹앱 URL 주소가 최종 생성됩니다.
형식은 기본적으로 https://script.google.com/macros/s/.../exec로 끝나게 됩니다. 이 연역 주소를 안전하게 복사해 둡니다.

5

웹 소스코드 설문 웹앱에 붙여넣기

1. 개발 환경 내에 있는 리액트 프로젝트 소스파일 중 /src/App.tsx 파일을 열어줍니다.
2. 약 27번째 줄 부근에 위치한 const SCRIPT_URL 상수를 찾아 큰따옴표 사이에 아까 복사해 둔 주소를 아래 예시처럼 채워 넣고 저장합니다.

// src/App.tsx
const SCRIPT_URL = "https://script.google.com/macros/s/AKfycb.../exec";
🚀 이제 설문조사 사이트를 새로고침하여 데이터를 입력하고 [공식 제출]을 클릭하시면 실시간으로 자동 정렬 배열되어 기입 완결됩니다!

3. 🚨 데이터가 전송되지 않을 때 자가진단 (트러블슈팅)

가장 유력 ① Google Apps Script 배포 설정 및 접근 제한 문제

계정 권한 설정을 생략하거나 잘못 선택하여 외부 시트 연동 명령 패키지가 누적 차단되는 문제입니다.

🛠️ 즉각적인 조치안:

1. Apps Script 브라우저 상단 오른쪽에 마련된 [배포] ➔ [배포 관리]를 클릭합니다.

2. 현재 등록된 활성 버전을 찾고 우측 상단부의 편집 아이콘(연필 모양) 단추를 클릭합니다.

3. "Who has access (액세스 권한이 있는 사용자)" 항목을 무조건 Anyone (모든 사용자)로 수정합니다.

4. 변경을 적용한 이후 버전 필드를 꼭 [새 버전 (New Version)]으로 올린 뒤 [배포]를 클릭해 새로운 주소 URL을 복사하여 리액트에 적용하십시오.

원인 추적 ② 라이브 앱스 스크립트 실행 분석 (Executions Log) 보기

코드 어딘가 실패가 났을 때 구글에서 제공하는 친절한 한글 오류창 내역을 대조하는 지름길입니다.

🛠️ 실행 분석 진단법:

1. 구글 Apps Script 편집기 좌측에 위치한 세로 도구바의 [실행 / Execution (시간표 원반 모양 무늬 아이콘)]을 클릭합니다.

2. 유저 제출로 호출된 doPost의 시간대별 기록과 트립 추적 상태가 정렬되어 나타납니다.

3. 오류 혹은 실패(Failed)로 표시된 붉은색 레코드를 클릭해 상세 설명을 확인합니다.

4. 만약 Exception: 스프레드시트를 열 수 없습니다. 라는 경고창이 있다면 스크립트 내부에 기재된 스프레드시트 고유 ID 소유주 권한 혹은 주소 오타를 조치하십시오.

필수 기본 ③ 스크립트 에디터 누적 저장 및 버전 업데이트 누락

단순히 에디터 내부에서 코드를 바꾼 후 [저장(컨트롤+S)]한 것만으로는 변경 사양이 라이브 웹에 반영되지 않는 구글의 기본적 패키지 성격에 따른 문제입니다.

🛠️ 최종 활성화 방법:

코드를 수정한 뒤에는 반드시 배포를 새로 해 주셔야 합니다. [배포] ➔ [새 배포]에 들러 기존 셋팅 요건을 유지한 채 새 버전으로 생성하기를 클릭해 완결해야만 라이브 포트폴리오에 적용됩니다.
